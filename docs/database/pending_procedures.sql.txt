==== PROCESAR_MOVITRAN (SQL_STORED_PROCEDURE) ====
CREATE  PROCEDURE [dbo].[PROCESAR_MOVITRAN] ( 	@strXML VARCHAR(MAX), 	@strProceso VARCHAR(50), 	@CrearArticulo VARCHAR(2) ) AS BEGIN 	IF @CrearArticulo IS NULL 	BEGIN 		SET @CrearArticulo = 'NO' 	END    	DECLARE @ProcesoInterno VARCHAR;	     DECLARE @iDoc int; 	SET @strXML = REPLACE(@strXML,'<xml xmlns:rs="urn:schemas-microsoft-com:rowset" xmlns:z="#RowsetSchema">','<?xml version="1.0" encoding="ISO-8859-1"?>'); 	SET @strXML = REPLACE(@strXML,'rs:data','root'); 	SET @strXML = REPLACE(@strXML,'z:row','MOVITRAN'); 	SET @strXML = REPLACE(@strXML,'</xml>','');     EXECUTE sp_xml_preparedocument @iDoc OUTPUT, @strXML; 	 	IF @strProceso = 'REMISION' 	BEGIN 		SELECT DatosXML.CodigoBarra, INVENTARIO.Nombre, DatosXML.Cantidad, INVENTARIO.Referencia, INVENTARIO.CodigoMarca, INVENTARIO.PrecioDetal, INVENTARIO.CostoPromedio 		FROM OpenXML(@iDoc, '/root/MOVITRAN', 2) 		WITH (CodigoBarra VARCHAR(15) '@CODBARRA', 			  Nombre VARCHAR(60) '@NOMB',  			  Cantidad NUMERIC(8,0) '@CANT') DatosXML INNER JOIN INVENTARIO ON INVENTARIO.CodigoBarra = DatosXML.CodigoBarra 		WHERE INVENTARIO.CostoPromedio > 0 AND PrecioDetal > 0 AND Status <> 0;  		SELECT DatosXML.CodigoBarra + ' ' + LTRIM(RTRIM(DatosXML.Nombre)) AS Prenda, 'Articulo No Existe' AS Mensaje 		FROM OpenXML(@iDoc, '/root/MOVITRAN', 2) 		WITH (CodigoBarra VARCHAR(15) '@CODBARRA', 			  Nombre VARCHAR(60) '@NOMB',  			  Cantidad NUMERIC(8,0) '@CANT') DatosXML LEFT OUTER JOIN INVENTARIO ON INVENTARIO.CodigoBarra = DatosXML.CodigoBarra 		WHERE INVENTARIO.Nombre IS NULL 		UNION 		SELECT DatosXML.CodigoBarra + ' ' + LTRIM(RTRIM(DatosXML.Nombre)) AS Prenda,  			   CASE WHEN INVENTARIO.CostoPromedio <= 0 THEN 'Costo Promedio del Articulo No Valido' 					ELSE CASE WHEN INVENTARIO.PrecioDetal <= 0 THEN 'Precio Detal del Articulo No Valido'  					ELSE CASE WHEN INVENTARIO.Status = 0 THEN 'Articulo No Activo' 			   END  END  END AS Mensaje	 		FROM OpenXML(@iDoc, '/root/MOVITRAN', 2) 		WITH (CodigoBarra VARCHAR(15) '@CODBARRA', 			  Nombre VARCHAR(60) '@NOMB',  			  Cantidad NUMERIC(8,0) '@CANT') DatosXML INNER JOIN INVENTARIO ON INVENTARIO.CodigoBarra = DatosXML.CodigoBarra 		WHERE INVENTARIO.CostoPromedio <= 0 OR PrecioDetal <= 0 OR Status = 0; 	END    	----------------------------------------------------------------------------- 	------------------------- CREACION DE ARTICULOS ----------------------------- 	----------------------------------------------------------------------------- 	IF @strProceso = 'COMPRAS' OR @strProceso = 'IMPORTAR_TRANSFERENCIA' 	BEGIN   		SELECT DatosXML.NumeroTransferencia, DatosXML.CodigoBarra, REPLACE(REPLACE(INVENTARIO.Nombre, CHAR(13), ''), CHAR(10), '') AS Nombre, DatosXML.Referencia, DatosXML.CodigoMarca, DatosXML.Cantidad 		FROM OpenXML(@iDoc, '/root/MOVITRAN', 2) 		WITH (NumeroTransferencia NUMERIC(18,0) '@NUMT',  			  CodigoBarra VARCHAR(15) '@CODBARRA', 			  Nombre VARCHAR(60) '@NOMB',  			  Cantidad NUMERIC(8,0) '@CANTI', 			  Referencia VARCHAR(10) '@REFER',  			  CodigoMarca VARCHAR(3) '@MARC') DatosXML INNER JOIN INVENTARIO ON INVENTARIO.CodigoBarra = DatosXML.CodigoBarra;   		------------------------------------------------------------------------- 		---------------------- ACTUALIZACION DE MARCAS -------------------------- 		------------------------------------------------------------------------- 		IF ((@CrearArticulo = 'SI') AND (@strProceso = 'COMPRAS')) OR (@strProceso = 'IMPORTAR_TRANSFERENCIA') 		BEGIN 			SET @ProcesoInterno = 'MARCAS';	 			UPDATE MARCAS 			SET Codigo = DatosXML.CodigoMarca, Nombre = DatosXML.NombreMarca, Status = 1 			FROM MARCAS INNER JOIN ( 					SELECT CodigoMarca, NombreMarca FROM OpenXML(@iDoc, '/root/MOVITRAN', 2) 					WITH (CodigoMarca VARCHAR(3) '@MARC', NombreMarca VARCHAR(20) '@NOMBMARCA') DatosXML 					GROUP BY CodigoMarca, NombreMarca 			) DatosXML ON DatosXML.CodigoMarca = MARCAS.Codigo  			INSERT MARCAS 			SELECT CodigoMarca, NombreMarca, 1 AS Status  			FROM OpenXML(@iDoc, '/root/MOVITRAN', 2)  			WITH (CodigoMarca VARCHAR(3) '@MARC', NombreMarca VARCHAR(20) '@NOMBMARCA') DatosXML 				LEFT JOIN MARCAS ON MARCAS.Codigo = DatosXML.CodigoMarca 			WHERE MARCAS.Codigo IS NULL 			GROUP BY CodigoMarca, NombreMarca; 		END   		------------------------------------------------------------------------- 		---------------------- ACTUALIZACION DE TALLAS -------------------------- 		------------------------------------------------------------------------- 		IF ((@CrearArticulo = 'SI') AND (@strProceso = 'COMPRAS')) OR (@strProceso = 'IMPORTAR_TRANSFERENCIA')  		BEGIN 			SET @ProcesoInterno = 'TALLAS'; 			UPDATE TALLAS 			SET Codigo = DatosXML.Talla 			FROM TALLAS INNER JOIN ( 				SELECT Talla FROM OpenXML(@iDoc, '/root/MOVITRAN', 2) 				WITH (Talla VARCHAR(3) '@TALLA') DatosXML 				GROUP BY Talla 			) DatosXML ON DatosXML.Talla = TALLAS.Codigo  			INSERT INTO TALLAS 			SELECT Talla FROM OpenXML(@iDoc, '/root/MOVITRAN', 2) 			WITH (Talla VARCHAR(3) '@TALLA') DatosXML LEFT JOIN TALLAS ON TALLAS.Codigo = DatosXML.Talla 			WHERE TALLAS.Codigo IS NULL 			GROUP BY Talla 		END  		------------------------------------------------------------------------- 		---------------------- ACTUALIZACION DE COLORES    ---------------------- 		------------------------------------------------------------------------- 		IF ((@CrearArticulo = 'SI') AND (@strProceso = 'COMPRAS')) OR (@strProceso = 'IMPORTAR_TRANSFERENCIA')  		BEGIN 			SET @ProcesoInterno = 'COLORES'; 			UPDATE COLORES 			SET Codigo = DatosXML.CodigoColor, Nombre = DatosXML.NombreColor, Status = 1 			FROM COLORES INNER JOIN ( 				SELECT CodigoColor, NombreColor FROM OpenXML(@iDoc, '/root/MOVITRAN', 2) 				WITH (CodigoColor VARCHAR(3) '@COLOR', NombreColor VARCHAR(30) '@NOMBCOLOR') DatosXML 				GROUP BY CodigoColor, NombreColor 			) DatosXML ON COLORES.Codigo = DatosXML.CodigoColor  			INSERT INTO COLORES 			SELECT CodigoColor, NombreColor, 1 AS Status FROM OpenXML(@iDoc, '/root/MOVITRAN', 2) 			WITH (CodigoColor VARCHAR(3) '@COLOR', NombreColor VARCHAR(30) '@NOMBCOLOR') DatosXML 			LEFT JOIN COLORES ON COLORES.Codigo = DatosXML.CodigoColor 			WHERE COLORES.Codigo IS NULL 			GROUP BY CodigoColor, NombreColor 		END  		------------------------------------------------------------------------- 		---------------------- ACTUALIZACION DE FABRICANTES --------------------- 		------------------------------------------------------------------------- 		IF ((@CrearArticulo = 'SI') AND (@strProceso = 'COMPRAS')) OR (@strProceso = 'IMPORTAR_TRANSFERENCIA') 		BEGIN 			SET @ProcesoInterno = 'FABRICANTES'; 			UPDATE FABRICANTES  			SET Codigo = DatosXML.CodigoFabricante, Nombre = DatosXML.NombreFabricante, Status = 1 			FROM FABRICANTES INNER JOIN 			(  				SELECT CodigoFabricante, NombreFabricante FROM OpenXML(@iDoc, '/root/MOVITRAN', 2) 				WITH (CodigoFabricante VARCHAR(12) '@FABRICANTE', NombreFabricante VARCHAR(50) '@NOMBFABRIC') DatosXML 				GROUP BY CodigoFabricante, NombreFabricante 			) DatosXML ON FABRICANTES.Codigo = DatosXML.CodigoFabricante  			INSERT INTO FABRICANTES 			SELECT CodigoFabricante, NombreFabricante, 1 AS Status FROM OpenXML(@iDoc, '/root/MOVITRAN', 2) 			WITH (CodigoFabricante VARCHAR(12) '@FABRICANTE', NombreFabricante VARCHAR(50) '@NOMBFABRIC') DatosXML 			LEFT JOIN FABRICANTES ON FABRICANTES.Codigo = DatosXML.CodigoFabricante 			WHERE FABRICANTES.Codigo IS NULL 			GROUP BY CodigoFabricante, NombreFabricante 		END   		------------------------------------------------------------------------- 		---------------------- ACTUALIZACION DE CATEGORIAS ---------------------- 		------------------------------------------------------------------------- 		IF ((@CrearArticulo = 'SI') AND (@strProceso = 'COMPRAS')) OR (@strProceso = 'IMPORTAR_TRANSFERENCIA') 		BEGIN 			IF @strProceso = 'IMPORTAR_TRANSFERENCIA'  			BEGIN  				SET @ProcesoInterno = 'CATEGORIAS'; 				PRINT 'Actualizando categorias...' 				UPDATE CATEGORIAS 				SET Status = 1 				FROM CATEGORIAS INNER JOIN ( 				SELECT CodigoCategoria, NombreCategoria FROM OpenXML(@iDoc, '/root/MOVITRAN', 2) 				WITH (CodigoCategoria VARCHAR(6) '@CATEGORIA', NombreCategoria VARCHAR(60) '@NOMBCATEG') DatosXML 				GROUP BY CodigoCategoria, NombreCategoria 				) DatosXML ON CATEGORIAS.Codigo = SUBSTRING(RTRIM(LTRIM(DatosXML.CodigoCategoria)),1,2)  				INSERT INTO CATEGORIAS 				SELECT CodigoCategoria, NombreCategoria, 1 AS Status FROM OpenXML(@iDoc, '/root/MOVITRAN', 2) 				WITH (CodigoCategoria VARCHAR(6) '@CATEGORIA', NombreCategoria VARCHAR(60) '@NOMBCATEG') DatosXML 				LEFT JOIN CATEGORIAS ON RTRIM(LTRIM(CATEGORIAS.Codigo)) = RTRIM(LTRIM(DatosXML.CodigoCategoria)) 				WHERE CATEGORIAS.Nombre IS NULL 				GROUP BY CodigoCategoria, NombreCategoria; 			END  		END  	END 	 	------------------------------------------------------------------------- 	---------------------- CREAR ARTICULOS PARA COMPRA ---------------------- 	-------------------------------------------------------------------------  	IF @strProceso = 'COMPRAS' 	BEGIN 		IF @CrearArticulo = 'SI' 		BEGIN   			SET @ProcesoInterno = 'CREACION DE NUEVAS PRENDAS';	 			INSERT INTO INVENTARIO(CodigoBarra,Referencia,CodigoMarca,Nombre,Talla,CodigoColor,Categoria,  								   PrecioDetal,PrecioMayor,PrecioAfiliado,Promocion,PrecioPromocion, 								   FechaInicial,FechaFinal, 								   CostoInicial,CostoPromedio,UltimoCosto, 								   PuntoReorden,ExistenciaInicial,Existencia, 								   FechaPrimerMovimiento,UltimaActualizacion,Fabricante,TipoImpuesto,Tipo,Nota,Status,CodigoBarraAnt) 			SELECT DatosXML.CodigoBarra, DatosXML.Referencia, DatosXML.CodigoMarca, DatosXML.Nombre, DatosXML.Talla, DatosXML.CodigoColor, '00',  			ISNULL((SELECT MAX(PrecioDetal) FROM INVENTARIO WHERE Referencia = RTRIM(LTRIM(DatosXML.Referencia)) AND CodigoMarca = RTRIM(LTRIM(DatosXML.CodigoMarca)) AND CodigoBarra <> RTRIM(LTRIM(DatosXML.CodigoBarra)) GROUP BY Referencia,CodigoMarca,Promocion), 0) AS PrecioDetal, 			ISNULL((SELECT MAX(PrecioMayor) FROM INVENTARIO WHERE Referencia = RTRIM(LTRIM(DatosXML.Referencia)) AND CodigoMarca = RTRIM(LTRIM(DatosXML.CodigoMarca)) AND CodigoBarra <> RTRIM(LTRIM(DatosXML.CodigoBarra)) GROUP BY Referencia,CodigoMarca,Promocion), 0) AS PrecioMayor,  			ISNULL((SELECT MAX(PrecioAfiliado) FROM INVENTARIO WHERE Referencia = RTRIM(LTRIM(DatosXML.Referencia)) AND CodigoMarca = RTRIM(LTRIM(DatosXML.CodigoMarca)) AND CodigoBarra <> RTRIM(LTRIM(DatosXML.CodigoBarra)) GROUP BY Referencia,CodigoMarca, Promocion), 0) AS RTRIM(LTRIM(DatosXML.CodigoBarra)) GROUP BY Referencia,CodigoMarca,Promocion), 0) AS Promocion, 			ISNULL((SELECT MAX(PrecioPromocion) FROM INVENTARIO WHERE Referencia = RTRIM(LTRIM(DatosXML.Referencia)) AND CodigoMarca = RTRIM(LTRIM(DatosXML.CodigoMarca)) AND CodigoBarra <> RTRIM(LTRIM(DatosXML.CodigoBarra)) GROUP BY Referencia,CodigoMarca,Promocion), 0) AS RTRIM(LTRIM(DatosXML.CodigoBarra)) GROUP BY Referencia,CodigoMarca,Promocion), 0) AS CostoInicial,  			ISNULL((SELECT MAX(CostoPromedio) FROM INVENTARIO WHERE Referencia = RTRIM(LTRIM(DatosXML.Referencia)) AND CodigoMarca = RTRIM(LTRIM(DatosXML.CodigoMarca)) AND CodigoBarra <> RTRIM(LTRIM(DatosXML.CodigoBarra)) GROUP BY Referencia,CodigoMarca,Promocion), 0) AS CostoPromedio,  			ISNULL((SELECT MAX(UltimoCosto) FROM INVENTARIO WHERE Referencia = RTRIM(LTRIM(DatosXML.Referencia)) AND CodigoMarca = RTRIM(LTRIM(DatosXML.CodigoMarca)) AND CodigoBarra <> RTRIM(LTRIM(DatosXML.CodigoBarra)) GROUP BY Referencia,CodigoMarca,Promocion), 0) AS UltimoCosto,  			0 AS PuntoReorden, 0 AS ExistenciaInicial, 0 AS Existencia, GETDATE() AS FechaPrimerMovimiento, GETDATE() AS UltimaActualizacion, CodigoFabricante AS Fabricante, 			1 AS TipoImpuesto, 0 AS Tipo, '' AS Nota, 1 AS Status, DatosXML.CodigoBarra AS CodigoBarraAnt 			FROM OpenXML(@iDoc, '/root/MOVITRAN', 2) 			WITH (CodigoBarra VARCHAR(15) '@CODBARRA', Nombre VARCHAR(60) '@NOMB', Referencia VARCHAR(10) '@REFER', 				  CodigoMarca VARCHAR(3) '@MARC', Talla VARCHAR(3) '@TALLA', CodigoColor VARCHAR(3) '@COLOR', CodigoFabricante VARCHAR(12) '@FABRICANTE', 				  CodigoBarraAnt VARCHAR(15) '@NUMTRA') DatosXML 			LEFT JOIN INVENTARIO ON INVENTARIO.CodigoBarra = DatosXML.CodigoBarra   			WHERE INVENTARIO.CodigoBarra IS NULL 			GROUP BY DatosXML.CodigoBarra, DatosXML.Referencia, DatosXML.CodigoMarca, DatosXML.Nombre, DatosXML.Talla, DatosXML.CodigoColor, DatosXML.CodigoFabricante; 		END   		SELECT DatosXML.* 		FROM OpenXML(@iDoc, '/root/MOVITRAN', 2) 		WITH (NumeroTransferencia NUMERIC(18,0) '@NUMT',  				CodigoBarra VARCHAR(15) '@CODBARRA', 				Nombre VARCHAR(60) '@NOMB',  				Cantidad NUMERIC(8,0) '@CANTI', 				Referencia VARCHAR(10) '@REFER',  				CodigoMarca VARCHAR(3) '@MARC') DatosXML  	END   	   	IF @strProceso = 'IMPORTAR_TRANSFERENCIA' 	BEGIN 		BEGIN TRY   			------------------------------------------------------------------------- 			----------------- VERIFICAR SUCURAL Y CASA MATRIZ ----------------------- 			------------------------------------------------------------------------- 			SET @ProcesoInterno = 'SUCURSAL Y CASA MATRIZ'; 			DECLARE @IdEmpresa VARCHAR(12); 			DECLARE @CodigoEmpresa VARCHAR(12); 			DECLARE @ActualizarPreciosArchivosExternos INT; 			SELECT @CodigoEmpresa = Codigo, @IdEmpresa = IdEmpresa, @ActualizarPreciosArchivosExternos = ActualizarPreciosArchivosExternos  			FROM PARAMETROS;  			IF NOT EXISTS(SELECT Codigo FROM SUCURSALES WHERE Codigo = @CodigoEmpresa) 			BEGIN 				INSERT INTO SUCURSALES (Codigo, Nombre, Direccion, Telefono, Status) 				SELECT @CodigoEmpresa, Nombre, Direccion, Telefono, 1 AS Status FROM PARAMETROS; 			END  			IF NOT EXISTS(SELECT  Codigo FROM SUCURSALES WHERE Codigo = @IdEmpresa) 			BEGIN 				INSERT INTO SUCURSALES (Codigo, Nombre, Direccion, Telefono, Status) 				VALUES (@IdEmpresa, 'CASA MATRIZ', '', '', 1) 			END  			------------------------------------------------------------------------- 			-------------------- PROCESAR TRANSFERENCIAS ---------------------------- 			------------------------------------------------------------------------- 			SET @ProcesoInterno = 'TRANSFERENCIA'; 			DECLARE @NumeroTransferencia NUMERIC(18,0) 			DECLARE @Fecha DATETIME 			DECLARE @CodigoEnvia VARCHAR(12) 			DECLARE @CodigoRecibe VARCHAR(12) 			DECLARE @TotalValor NUMERIC(18,2) 			DECLARE
